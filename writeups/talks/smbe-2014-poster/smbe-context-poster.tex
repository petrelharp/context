\headerbox{Context-dependence}{name=intro,column=0,row=0,span=1}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

  \textbf{Data:} Long (nucleotide) sequences separated by (evolutionary) time
    with known (phylogenetic) relationship.


  \textbf{Process:} Sequence substitution rates depend on local context.


  \textbf{Goal:} Infer substitution motifs;
    mutation rates;
    relative amounts of time.

  \textbf{Background:} 
  The raw mutational spectrum is highly context-dependent and multinuleotide 
  \citep{schaibley2013influence,schrider2011pervasive,terekhanova2013prevalence,harris2013errorprone}.
  This is an issue for methods in phylogenetics and ancestral sequence reconstruction
  that assume independence of sites.
  Previous  work --
  \citet{siepel2004phylogenetic} assumed Markov dependency along the sequence;
  \citet{lunter2004nucleotide} used a matrix decomposition;
  and \citet{pedersen2000dependent,hobolth2008markov} used data augmentation with an MCMC algorithm
  -- is not computationally feasible for long sequences.
  Similar: Markov random fields in image reconstruction \citep{besag1986statistical};
  inference for cellular automata models \citep{clarke1994cellular,zorzenondossantos2001dynamics}.

}

\headerbox{General mutational grammar}{name=grammar,column=0,span=1,below=example}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


  \textbf{Mutation} rules $(a,b,r)$: 
    locations matching $a$ mutate to $b$ at rate $r$ 
  \[
  a \longrightarrow b \qquad \text{ at rate } r
  \]
  $a$, $b$ can be \emph{any} patterns of the same length.

    Also: fixation bias (e.g.\ GC-bias).

%   \textbf{Fixation} function: 
%     mutations take effect with probability 
%     \begin{align*}
%       f(s(\text{new}) - s(\text{old})) ,
%     \end{align*}
%     e.g.\ $f(ds) = N_e (1-e^{-2 ds})/(1-e^{-2 N_e ds})$
% 
%   \textbf{Selection} is additive (and abstract):
%     given rules $(a_i,s_i)$,
%     \begin{align*}
%       s(x) = \sum_i s_i n_i(x),
%     \end{align*}
%     with $n_i(x)$ number of times $a_i$ matches in $x$


\textbf{Assuming:}
    homogeneity along the sequence and through time

    \textbf{Not assuming:}
    stationarity;
    reversibility.

    \emph{Example:} GC-content need not be at equilibrium.



}

\headerbox{Exact solution}{name=exact,column=0,span=1,below=grammar}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Say $\lambda$ is the \emph{total mutation rate}
and $P$ is the matrix of single-mutation probabilities, i.e.
\begin{align}
  \P\{ x \longrightarrow y \text{ in time $dt$ }\} = P_{xy} \lambda dt 
\end{align}
e.g.\ $P_{ACA \to ATA} = \mu_{CT}$ and $P_{GAC \to TCG} = 0$.
Then the transition probabilities are
\begin{align}
  & \P\{ x \longrightarrow y \text{ in time $t$ }\} = \sum_{k \ge 0} e^{-\lambda t} \frac{ (\lambda t)^k }{ k! } (P^k)_{xy} \\
  &\qquad = \sum_{k \ge 0} \text{( prob there are $k$ mutations )} \times \text{ ( prob k mutations takes $x \to y$ ) } .
\end{align}

$P$ {also equals} $e^{tG}$ with $G_{xy} = \lambda P_{xy}$ and $G_{xx} = - \sum_{y\neq x} G_{xy}$.
For length-$n$ sequences, \textbf{$P$ is $4^n \times 4^n$}.

% \[ % max rate < 1 :
% e^{tG}  = e^{t((G+I)-I)} = e^{-t} I e^{t(G+I)} = \sum_k e^{-t} t^k/k! (G+I)^k
% \]
% If $G$ is the \emph{rate matrix}, i.e. for $x \neq y$,
% \begin{align}
%   G_{xy} &= ( \text{ rate at which sequence $x$ changes to $y$ } ) 
% \end{align}
% and $G_{xx} = - \sum_{y\neq x} G_{xy}$, then


}

\headerbox{The relevant fringe}{name=windows,column=0,span=1,below=exact}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

    \begin{itemize}
      \item If the size of the maximum context is $k$
      \item and the expected number of such mutations per site is $\mu t$
      \item a site is unlikely to be affected by context further way than, say, $k^2 \mu t$.
      \item i.e.\ the probability of a particular transition at a site is almost unaffected by changes further away.
    \end{itemize}

\headerbox{Computation}{name=computation,column=1}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Tricks:} Krylov methods to compute $e^{tG}v$ for sparse $G$ (also: gamma-distributed times);\\
precompute mapping of parameters to $G$ to make parameter update quick.


  \vspace{1em}

  \textbf{Choices:}
  \begin{itemize}
    \item Long ($w$) and short ($r$) pattern sizes
    \item Set of short patterns: maximize information
  \end{itemize}

  \vspace{1em}

  \textbf{Memory usage:} constant times the (pattern count) data.

  \vspace{1em}

  \textbf{Computational time:} $O(w 4^{w} (\#\text{short patterns}))$ operations 
  \begin{itemize}
    \item $(w=5,r=1) \longrightarrow$  a second
    \item $(w=8,r=2) \longrightarrow$  a minute
  \end{itemize}

  \vspace{1em}

  \textbf{Feasible:} window sizes, nearest neighbor context \\
  \hspace{3em} $w=4$: mean density of mutations ${} < 0.5$ per site\\
  \hspace{3em}   $w=8$: mean density of mutations ${} < 2$ per site


}
